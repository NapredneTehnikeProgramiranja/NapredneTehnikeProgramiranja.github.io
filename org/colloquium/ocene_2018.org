  (defun ntp/mul-const (column factor)
    "Multiply COLUMN by constant FACTOR. Result is normalized to
  100."
    (if (not (string= "" column))
        (let ((result (* (string-to-number column)
                         factor)))
          (if (> result 100)
              100
            (format "%0.2f" result)))
      ""))
